\section{\sffamily Accumulator}

\begin{RTLexample}[Accumulator Data Path]
\begin{tikzpicture}[>=Stealth, thick, scale=0.8]
    % 1. Functional Elements
    \node[rtl_op] (Add) at (0,0) {+};
    \node[rtl_reg_clk, right=2cm of Add] (Reg) {Acc};
    
    % 2. Input Data Path
    \draw[<-, rtl_bus={16}] (Add.north) -- ++(0, 1.2) -- ++(-2, 0) node[left] {Data\_In};
    
    % 3. Forward Path (Adder to Register)
    \draw[->, rtl_bus={16}] (Add.east) -- (Reg.west) node[midway, above] {sum};
    
    % 4. Feedback Path (Register to Adder)
    % We use routeU: Start at east, move 1cm out, 2.5cm down, and end at Add.south
    \routeU[->, rtl_bus={16}, blue]{Reg.east}{1}{-2.5}{Add.west};
    
    % 5. Control Signals
    \draw[<-] (Reg-CLK) -- ++(-1, 0) node[below] {clk};
    
    % 6. Output
    \node[circ, right=0.75 of Reg.east] (n1){};
    \draw[->, rtl_bus={16}] (n1) -- ++(2,0) node[right] {Data\_Out};
\end{tikzpicture}
\end{RTLexample}

\begin{RTLexample}[CORDIC Core Datapath]
\begin{tikzpicture}[>=Stealth, thick, scale=0.7, transform shape]
    % 1. Controller (FSM)
    \node[rtl_module={Controller}, minimum width=3cm] (Ctrl) at (4, 6.5) {FSM State: \texttt{COMPUTE}};

    % 2. Pipeline Registers (xREG, yREG, zREG)
    \node[rtl_reg_clk, is_cutset=red] (xReg) at (0, 3) {xREG};
    \node[rtl_reg_clk, is_cutset=red] (yReg) at (0, 0) {yREG};
    \node[rtl_reg_clk, is_cutset=red] (zReg) at (0,-3) {zREG};

    % 3. Shifters (shiftedX, shiftedY)
    \node[rtl_shifter, right=1.5cm of xReg] (shX) {$\gg i$};
    \node[rtl_shifter, right=1.5cm of yReg] (shY) {$\gg i$};

    \createMux[right=2cm of shX]{xPreAdd}{{0}, {1}};
    \createMux[right=2cm of shY]{yPreAdd}{{0}, {1}};

    % 4. Arithmetic Operators (ALU Logic)
    %\node[rtl_op, right=1.5cm of shX] (addX) {$\pm$};
    %\node[rtl_op, right=1.5cm of shY] (addY) {$\pm$};
    %\node[rtl_op, right=1.5cm of zReg] (addZ) {$\pm$};

    % 5. LUT (ATAN / ATANH)
    \node[rtl_module={LUT}, below=2.5cm of addZ] (LUT) {atanh / atan / shift};

    % --- Routing ---

    % Data Path Connections
    \draw[->, rtl_bus={32}] (xReg) -- (shX);
    \draw[->, rtl_bus={32}] (yReg) -- (shY);
    
    % Cross-over for CORDIC (x helps y, y helps x)
    \draw[->, rtl_bus={32}] (shX.east) -- (addY.west);
    \draw[->, rtl_bus={32}] (shY.east) -- (addX.west);
    \draw[->, rtl_bus={32}] (zReg.east) -- (addZ.west);
    \draw[->, rtl_bus={32}] (LUT.north) -- (addZ.south) node[midway, left, xshift=-0.2cm] {$\alpha_i$};

    \draw[->, dashed, line width=2] (Ctrl.south) -- ++(0,-1) -| (shX.north) node[pos=0.8, right, xshift=0.2cm] {shift $\mathsf{i}$};
    \draw[->, dashed] (Ctrl.south) -- ++(0,-1) -| (addX.north) node[pos=0.8, right, xshift=0.2cm] {$\mathsf{d}$};
    \draw[->, dashed, line width=2] (shX.south) -- (shY.north) node[pos=0.8, right, xshift=0.2cm] {shift $\mathsf{i}$};
